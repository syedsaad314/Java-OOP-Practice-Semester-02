Note: The array should be sorted first.
import java.util.Scanner;

public class Main {

    public static void main(String[] args) {
        int[] array = new int[100];
        int target;
        System.out.print("Enter target:");
        Scanner sc = new Scanner(System.in);
        target = sc.nextInt();
        for(int i=0; i<array.length; i++) {
            array[i] = i+1;
        }

        int index = binarySearch(array, target);  //will find the index at which target is located

        if(index == -1)
            System.out.println("Target not found");
        else
            System.out.println("Target found at index: " + index);
    }

    public static int binarySearch(int[] array, int target) {
        int low=0, high=array.length-1;

        while(low <= high) {
            int middle = low + (high - low)/2;
            int value = array[middle];

            //System.out.println("Middle: " + value); (to see steps)

            if (value<target)
                low = middle+1;  //check above 50
            else if (value>target)
                high = middle-1;  //check below 50
            else
                return middle;  //in case target found
        }

        return -1;  //in case target not found
    }
}
